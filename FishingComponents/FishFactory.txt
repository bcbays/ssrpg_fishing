var loot = import Games/Fishing2/FishingComponents/FishLootTable

func MakeFish(min, max)
  var fish = new Games/Fishing2/FishingComponents/FishData

  var selectedFish = RngRange(min, max)

  ?selectedFish >= loot.selMin0 & selectedFish < loot.selMax0
    fish.name = te.xt(tid_fish_speedyfin)
    fish.art = ascii
#########
#><))8>##
asciiend
    fish.rodPivotX = 7
    fish.rodPivotY = 0
    fish.height = 3
    fish.scalesMin = loot.scalesMin0
    fish.scalesMax = loot.scalesMax0
    fish.acceleration = 15


  :?selectedFish >= loot.selMin1 & selectedFish < loot.selMax1
    fish.name = te.xt(tid_fish_bowfish)
    fish.art = ascii
#.  )  ##
# )( o>##
#´  )´ ##
asciiend
    fish.rodPivotX = 7
    fish.rodPivotY = 0
    fish.height = 3
    fish.scalesMin = loot.scalesMin1
    fish.scalesMax = loot.scalesMax1
    fish.acceleration = 16


  :?selectedFish >= loot.selMin2 & selectedFish < loot.selMax2
    fish.name = te.xt(tid_fish_spookfin)
    fish.art = ascii
#  (°_ ##
#_/_ o\##
#´  `'"##
asciiend
    fish.rodPivotX = 7
    fish.rodPivotY = 0
    fish.height = 3
    fish.scalesMin = loot.scalesMin2
    fish.scalesMax = loot.scalesMax2
    fish.acceleration = 17


  :?selectedFish >= loot.selMin3 & selectedFish < loot.selMax3
    fish.name = te.xt(tid_fish_poker)
    fish.art = ascii
##,\_####
≥´_ °`──-
##/´‾####
asciiend
    fish.rodPivotX = 6
    fish.rodPivotY = 0
    fish.height = 3
    fish.scalesMin = loot.scalesMin3
    fish.scalesMax = loot.scalesMax3
    fish.acceleration = 18


  :?selectedFish >= loot.selMin4 & selectedFish < loot.selMax4
    fish.name = te.xt(tid_fish_outchboi)
    fish.art = ascii
# .`'´,##
#); <°c##
# ´.,.`##
asciiend
    fish.rodPivotX = 7
    fish.rodPivotY = 0
    fish.height = 3
    fish.scalesMin = loot.scalesMin4
    fish.scalesMax = loot.scalesMax4
    fish.acceleration = 19

  :?selectedFish >= loot.selMin5 & selectedFish < loot.selMax5
    fish.name = te.xt(tid_fish_whopper)
    fish.art = ascii
##(‾‾`._#
(\,' ‾ o\
(:   >( <
(/`;-.,-´
###‾‾####
asciiend
    fish.rodPivotX = 9
    fish.rodPivotY = 1
    fish.height = 5
    fish.scalesMin = loot.scalesMin5
    fish.scalesMax = loot.scalesMax5
    fish.acceleration = 20

  :?selectedFish >= loot.selMin6 & selectedFish < loot.selMax6
    fish.name = te.xt(tid_fish_moonfin)
    fish.art = ascii
###.--.##
(./ o,-\#
#)) (   #
(´\  `-;#
###`--´##
asciiend
    fish.rodPivotX = 7
    fish.rodPivotY = 1
    fish.height = 5
    fish.scalesMin = loot.scalesMin6
    fish.scalesMax = loot.scalesMax6
    fish.acceleration = 21

  :
    fish.name = te.xt(tid_fish_angrish)
    fish.art = ascii
###__####
_.-. `. _
`.  `; ò/
-´_.´`,´#
####/´###
asciiend
    fish.rodPivotX = 8
    fish.rodPivotY = 1
    fish.height = 5
    fish.scalesMin = loot.scalesMin7
    fish.scalesMax = loot.scalesMax7
    fish.acceleration = 22


  fish.scales = RngRange(fish.scalesMin, fish.scalesMax)
  var weightD = (fish.scales / 10)
  fish.weight = weightD + "." + (fish.scales - weightD * 10)
  //>@fish.scalesMin@, @fish.scalesMax@, @fish.scales@; @weightD@, @fish.weight@

  return fish



func RngRange(min, max)
  var n
  n = min + rng % (max - min + 1)
  return n
